## ğŸ“Š DATABASE SCHEMA (Proposed)

```sql
-- Bills synced from Tally
CREATE TABLE bills (
    id INTEGER PRIMARY KEY,
    tally_voucher_id TEXT UNIQUE,
    voucher_number TEXT,
    voucher_type TEXT,  -- 'Sales', 'Credit Sales', 'Pending Sales Bill', 'A Pto Bill'
    voucher_date DATE,
    party_ledger TEXT,
    amount DECIMAL(15,2),
    payment_status TEXT DEFAULT 'pending',  -- 'pending', 'partial', 'paid'
    dispatch_status TEXT DEFAULT 'created', -- 'created', 'ready', 'dispatched', 'customer_taken'
    synced_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Receipts created from Dashboard
CREATE TABLE receipts (
    id INTEGER PRIMARY KEY,
    bill_id INTEGER REFERENCES bills(id),
    tally_voucher_id TEXT,
    amount DECIMAL(15,2),
    payment_mode TEXT,  -- 'cash', 'bank'
    created_by TEXT,
    synced_to_tally BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Sack/Bundle tracking
CREATE TABLE sacks (
    id INTEGER PRIMARY KEY,
    sack_number TEXT UNIQUE,
    customer_name TEXT,
    status TEXT DEFAULT 'packing',  -- 'packing', 'ready', 'dispatched'
    notes TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Items in sack (your bills + external vendor items)
CREATE TABLE sack_items (
    id INTEGER PRIMARY KEY,
    sack_id INTEGER REFERENCES sacks(id),
    bill_id INTEGER REFERENCES bills(id) NULL,  -- NULL for external vendor
    external_vendor TEXT,
    external_amount DECIMAL(15,2),
    item_description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Sync state tracking
CREATE TABLE sync_state (
    id INTEGER PRIMARY KEY,
    last_sync_time TIMESTAMP,
    last_voucher_date DATE,
    sync_status TEXT
);

-- User management
CREATE TABLE users (
    id INTEGER PRIMARY KEY,
    username TEXT UNIQUE,
    password_hash TEXT,
    role TEXT,  -- 'biller', 'cashier', 'manager'
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Audit log
CREATE TABLE audit_log (
    id INTEGER PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    action TEXT,
    table_name TEXT,
    record_id INTEGER,
    old_value TEXT,
    new_value TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```
